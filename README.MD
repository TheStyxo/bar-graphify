# Bar Graphify
### A simple and lightweight node module to create bar graphs as strings.

<br>

# Installation
**NPM** :

```bash
npm install bar-graphify
```
---
## Example usage:
```js
const Graph = require('bar-graphify');
const options = {
    xLabels: ["X1", "X2", "X3"],
    yLabels: ["Y1", "Y2", "Y3"],
    fillChar: "â–„",
    emptyChar: " ",
    cursorChar: "^",
    cursorLocation: 1,
    collumnWidth: 3,
    firstCollumnWidth: 4,
    collumnSeparator: " ",
    firstCollumnSeparator: "| ",
    fillHeights: [2, 1, 3]
}
const graph = new Graph(options);

console.log(graph.toString()); 
```
